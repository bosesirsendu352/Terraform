terraform {
  required_providers {
    aws = {
      source  = "hashicorp/aws"
      version = "~> 4.16"
    }
  }

  required_version = ">= 1.2.0"
}

provider "aws" {
  region  = var.region_name
}
# Creating bucket through terraform

resource "aws_s3_bucket" "my_bucket" {
  bucket = var.bucket_name
  acl    = "public"

  tags = {
    Name = var.bucket_name
  }
}
